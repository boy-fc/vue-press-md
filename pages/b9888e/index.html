    # vue-cli创建

    // 安装或者升级
npm install -g @vue/cli
// 保证vue cli 版本在 4.5.0 以上
vue --version
// 创建项目
vue create my-project

    # vite创建

    // 全局安装create-vite-app
npm i -g create-vite-app
// 创建项目目录
create-vite-app 项目名
// 运行
cd 项目名
npm install
npm run dev
